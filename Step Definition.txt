// ZeroWebSecurityPage.js

class ZeroWebSecurityPage {
  visit() {
    cy.visit('https://www.zerowebsecurity.com');
  }
  getNavigationMenu() {
    return cy.get('nav');
  }
  getHeading() {
    return cy.get('h1');
  }
  getFooter() {
    return cy.get('footer');
  }
  getCopyrightText() {
    return cy.contains('Copyright');
  }
}
export default new ZeroWebSecurityPage();
